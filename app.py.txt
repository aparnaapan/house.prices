import streamlit as st
import joblib

# Load the saved model
model = joblib.load('home_price_model.pkl')

# Streamlit UI
st.title("üè° Home Price Prediction App")
st.write("Enter the required details to predict the home price:")

# Example: taking area as input
area = st.number_input("Enter the Area in square feet", min_value=100, step=100)

# Add more inputs here if your model has more features

# Predict button
if st.button("Predict Price"):
    prediction = model.predict([[area]])  # Update if more features
    st.success(f"Predicted Home Price: ‚Çπ {round(prediction[0], 2)}")
